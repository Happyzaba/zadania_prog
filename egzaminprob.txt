class Program
{
    static void Main()
    {
        Console.Write("Ile kilometrów przebiegłeś: ");
        if (!int.TryParse(Console.ReadLine(), out int n) || n <= 0)
        {
            Console.WriteLine("Błąd danych.");
            return;
        }

        int[] czasy = new int[n];

        Console.WriteLine("Podaj czasy każdego kilometra (w sekundach):");
        for (int i = 0; i < n; i++)
        {
            Console.Write($"{i + 1}: ");
            if (!int.TryParse(Console.ReadLine(), out czasy[i]) || czasy[i] < 0)
            {
                Console.WriteLine("Błędna wartość czasu.");
                return;
            }
        }

        Console.Write("Podaj długość odcinka (w km): ");
        if (!int.TryParse(Console.ReadLine(), out int k) || k <= 0 || k > n)
        {
            Console.WriteLine("Błędna długość odcinka.");
            return;
        }

        var wynik = NajlepszyOdcinek(czasy, k);

        Console.WriteLine($"Najlepszy czas na {k}km odcinku wynosi: {wynik}");
    }

    /**
     * NajlepszyOdcinek
     * Funkcja oblicza najmniejszą sumę kolejnych k czasów (w sekundach) – czyli najlepszy odcinek.
     * Parametry wejściowe:
     * czasy - tablica czasów w sekundach
     * k - długość odcinka (w kilometrach)
     * wartość zwracana:
     * najmniejsza suma k kolejnych czasów
     * autor: <TWÓJ_NUMER>
     * Algorytm: sliding window (przesuwne okno)
     **/
    static int NajlepszyOdcinek(int[] czasy, int k)
    {
        int suma = 0;
        for (int i = 0; i < k; i++)
            suma += czasy[i];

        int min = suma;

        for (int i = k; i < czasy.Length; i++)
        {
            suma += czasy[i];
            suma -= czasy[i - k];
            if (suma < min)
                min = suma;
        }
        return min;
    }
}

using System;
using System.ComponentModel;
using System.Windows.Input;
using Microsoft.Maui.Controls;
using System.Globalization;

namespace SumatorCzasowApp
{
    public class MainViewModel : INotifyPropertyChanged
    {
        private string minutesInput;
        private string secondsInput;
        private int totalSeconds = 0;

        public event PropertyChangedEventHandler PropertyChanged;

        public string MinutesInput
        {
            get => minutesInput;
            set { minutesInput = value; OnPropertyChanged(nameof(MinutesInput)); }
        }

        public string SecondsInput
        {
            get => secondsInput;
            set { secondsInput = value; OnPropertyChanged(nameof(SecondsInput)); }
        }

        public string TotalLabel => "Łączny czas (m:s):";

        public string TotalTime
        {
            get
            {
                int m = totalSeconds / 60;
                int s = totalSeconds % 60;
                return $"{m}:{s:D2}";
            }
        }

        public ICommand DoliczCommand => new Command(OnDolicz);
        public ICommand ResetCommand => new Command(OnReset);

        private void OnDolicz()
        {
            // walidacja
            if (!int.TryParse(MinutesInput, NumberStyles.Integer, CultureInfo.InvariantCulture, out int m))
            {
                ShowError("Błędna wartość");
                return;
            }
            if (!int.TryParse(SecondsInput, NumberStyles.Integer, CultureInfo.InvariantCulture, out int s))
            {
                ShowError("Błędna wartość");
                return;
            }
            if (m < 0 || s < 0 || s >= 60)
            {
                ShowError("Błędna wartość sekund lub minut");
                return;
            }

            // dodaj
            totalSeconds += m * 60 + s;

            // wyczyść pola do kolejnego wpisu
            MinutesInput = string.Empty;
            SecondsInput = string.Empty;

            // powiadom UI
            OnPropertyChanged(nameof(TotalTime));
        }

        private async void ShowError(string message)
        {
            // proste wyświetlenie alertu - w MAUI można użyć Application.Current.MainPage.DisplayAlert
            await Application.Current.MainPage.DisplayAlert("błąd", message, "OK");
        }

        private void OnReset()
        {
            totalSeconds = 0;
            MinutesInput = string.Empty;
            SecondsInput = string.Empty;
            OnPropertyChanged(nameof(TotalTime));
        }

        private void OnPropertyChanged(string name) =>
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(name));
    }
}
