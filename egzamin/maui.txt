            <Label x:Name="LabelG" WidthRequest="40" Text="255"/>
        </StackLayout>

        <StackLayout Orientation="Horizontal">
            <Label Text="B" WidthRequest="20"/>

using Microsoft.Maui.Controls;

namespace WzornikRGB;

public partial class MainPage : ContentPage
{
    private int r = 255, g = 255, b = 255;

    public MainPage()
    {
        InitializeComponent();
        UpdateDuzyProstokat();
    }

    private void OnSliderChanged(object sender, ValueChangedEventArgs e)
    {
        r = (int)SliderR.Value;
        g = (int)SliderG.Value;
        b = (int)SliderB.Value;

        LabelR.Text = r.ToString();
        LabelG.Text = g.ToString();
        LabelB.Text = b.ToString();

        UpdateDuzyProstokat();
    }

    private void UpdateDuzyProstokat()
    {
        DuzyProstokat.Color = Color.FromRgb(r, g, b);
    }

    private void OnPobierzClicked(object sender, EventArgs e)
    {
        MalyProstokat.Color = Color.FromRgb(r, g, b);
        LabelRGB.Text = $"{r}, {g}, {b}";
    }
}
